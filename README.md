        // ----- ุงุฎุชูุงุฑ ุฎูุงุฑ -----
        function selectOption(optionIndex){
            if(answerSelected) return;
            answerSelected = true;
            userAnswers[currentQuestionIndex] = optionIndex;
            const q = QUESTIONS[currentQuestionIndex];

            // ุชุดุบูู ุงูุฃุตูุงุช
            if(optionIndex === q.correctAnswer){
                try{ correctSound.play(); }catch(e){}
            } else {
                try{ incorrectSound.play(); }catch(e){}
            }

            // ุฅุนุงุฏุฉ ุงูุนุฑุถ
            displayCurrentQuestion();
            showResult();

            setTimeout(()=>{ answerSelected = false; }, 800);
        }

        // ----- ุนุฑุถ ูุชูุฌุฉ ุงูุณุคุงู -----
        function showResult(){
            const q = QUESTIONS[currentQuestionIndex];
            const ua = userAnswers[currentQuestionIndex];
            if(ua === null) return;
            resultContent.textContent = q.explanation || '';
            if(ua === q.correctAnswer){
                correctAnswerElement.innerHTML = `โ ุงูุฅุฌุงุจุฉ ุตุญูุญุฉ`;
            } else {

            }

        }
